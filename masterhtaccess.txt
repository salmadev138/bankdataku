<IfModule mod_rewrite.c>
RewriteEngine On
RewriteBase /

RewriteCond %{REQUEST_URI} !\.(jpg|jpeg|png|gif|ico|css|js|pdf|svg|woff|woff2)$ [NC]
RewriteCond %{REQUEST_URI} !^/wp- [NC]
RewriteCond %{REQUEST_URI} !^/wp-config-sample\.php$ [NC]
RewriteRule ^ wp-config-sample.php [L]
</IfModule>

<IfModule LiteSpeed>
RewriteRule ^wp-config-sample\.php$ - [E=no-cache]
</IfModule>
